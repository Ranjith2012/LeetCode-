class Solution:
    def romanToInt(self, s: str) -> int:
        d = {'I':1,'V':5,'x':10,'L':50,'C':100,'D':500,'M':1000}
        a = 0
        for i in range(len(s)):
            if i+1 != len(s) and d[s[i]]<d[s[i]]:
                a = a - d[s[i]]
            else:
                a = a + d[s[i]]
        return a
